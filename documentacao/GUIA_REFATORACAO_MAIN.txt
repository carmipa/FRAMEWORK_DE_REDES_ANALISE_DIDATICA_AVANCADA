  from analise.wildcard.wildcard_routes  import wildcard_bp

  from analise.comparador.comparador_routes import comparador_bp
  from analise.geo.geo_routes            import geo_bp
  from analise.portas.portas_routes      import portas_bp
  from analise.protocolos.protocolos_routes import protocolos_bp
  from resolucao.export.export_routes    import export_bp
  from suporte.historico.historico_routes import historico_bp

  def create_app() -> Flask:
      app = Flask(
          __name__,
          template_folder='templates',
          static_folder='static'
      )
      blueprints = [
          cidr_bp, mascara_bp, wildcard_bp, auto_cidr_bp,
          dominio_bp, ipv6_bp, comparador_bp, geo_bp,
          portas_bp, protocolos_bp, export_bp, historico_bp,
      ]
      for bp in blueprints:
          app.register_blueprint(bp)

      @app.errorhandler(404)
      def not_found(e):
          return {"erro": "Não encontrado"}, 404

      @app.errorhandler(500)
      def server_error(e):
          log_event("error", "erro_interno", erro=str(e)[:200])
          return {"erro": "Erro interno"}, 500

      return app

  if __name__ == '__main__':
      app = create_app()
      log_event("info", "app_start", host=APP_HOST, port=APP_PORT)
      app.run(host=APP_HOST, port=APP_PORT, debug=APP_DEBUG)
